
#### Basics-1

    node{
        stage("stage-1"){
          println "From stage-1"
        }
        stage("stage-2"){
          println "From stage-1"
        }
        stage("stage-3"){
          println "From stage-1"
        }
        // we can define n number of stages
    }

    node{

            mkdir reportDir

          """
          
          sh '''
            echo "Multi line shell commands"

        }
